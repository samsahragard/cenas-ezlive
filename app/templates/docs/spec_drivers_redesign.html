3 · Route change (store_routes.py)

@store_bp.route("/drivers", methods=["GET"])
@requires_permission("drivers.admin")
def drivers_admin():
    # NEW: tab-filter via querystring. Defaults to 'active'.
    status = request.args.get("status", "active")
    if status not in ("active", "inactive"):
        status = "active"
    show_active = (status == "active")

    db = next(get_db())
    try:
        q = db.query(Driver).filter(Driver.active == show_active)
        if g.current_location != "both":
            q = q.filter(Driver.location == g.current_location)
        rows = q.order_by(Driver.location, Driver.name).all()

        # Counts for tab labels (compute BOTH so labels show magnitudes)
        active_count_q = db.query(Driver)
        inactive_count_q = db.query(Driver).filter(Driver.active == False)
        active_count_q = active_count_q.filter(Driver.active == True)
        if g.current_location != "both":
            active_count_q = active_count_q.filter(Driver.location == g.current_location)
            inactive_count_q = inactive_count_q.filter(Driver.location == g.current_location)
        active_count = active_count_q.count()
        inactive_count = inactive_count_q.count()

        # ... (rest of existing logic for latest_shift_for + verified_for stays as-is)

        return render_template(
            "driver_admin.html",
            drivers=rows,
            current_status=status,          # NEW
            active_count=active_count,      # NEW
            inactive_count=inactive_count,  # NEW
            # ... existing kwargs unchanged
        )
    finally:
        db.close()
